database_schema
database_name = "knights_tour"
tables = {
  user_information: {
    account_id: varchar(255) primary key unique not null [auto_generate],
    first_name: varchar(50) not null,
    surname: varchar(50) not null,
    email_addr: varchar(255) not null,
    username: varchar(255) not null,
    birthdate: date,
    password: varchar(255) [in bcrypt format],
    role: varchar(255) [player. admin]
  }
  play_history_metadata: {
    session_id: varchar(255) primary key,
    account_id: varchar(255) [from session],
    status: enum('Win', 'Loss', 'Abandoned') default 'Abandoned',
    total_moves: int(4),
    started_at: date,
    ended_at: date,
    board_size: varchar(10) [represent as x, y],
    foreign key(account_id) references user_information(account_id)
  }
  move_table: {
    move_id: int(5) primary key auto_increment,
    session_id: varchar(255) [references metadata],
    move_number: int(4),
    x_pos: tinyint
    y_pos: tinyiny
    created_at: date,
    foreign key (session_id) references play_history_metadata(session_id) on delete cascade, unique (session_id, move_number)
  }
}
session variables = {
  account_id: storeable on play_history_metadata
  session_id: auto_generated for every game initiated
  username: from account_information
}

input_names = {
  register: [acctid, fname, sname, email, usrnm, dob, cpass, isAdmin]
}
